📋 Kirim ke Google Sheets
⚙️ Mode Pengiriman:
• Append (Tambah) — Data baru ditambahkan di bawah data lama. Cocok untuk tracking harian, semua histori tersimpan.
• Overwrite (Timpa) — Sheet dikosongkan dahulu lalu diisi ulang. Cocok untuk snapshot terkini saja.
⚠️ Peringatan: Mode Overwrite akan menghapus semua data lama di sheet tujuan sebelum mengisi data baru. Histori sebelumnya tidak bisa dipulihkan!
📥 Load & Sinkronisasi Data dari Google Sheet
Data dari Google Sheet akan menggantikan data lokal browser ini. Gunakan ini untuk sinkronisasi antar laptop/device.
⚙️ Cara Setup Apps Script (cukup sekali):
1. Buka Google Sheet Anda → klik menu Extensions → Apps Script
2. Hapus semua kode yang ada → Paste script di bawah → klik 💾 Save
3. Klik Deploy → New deployment → pilih tipe Web App
4. Isi deskripsi bebas (misal: HRD Dashboard)
5. Execute as: Me  |  Who has access: Anyone
6. Klik Deploy → izinkan akses → salin URL yang muncul
7. Paste URL tersebut ke tab "Kirim Data" di atas → klik Kirim

⚠️ Jika pernah deploy sebelumnya, pilih Manage deployments → edit → New version

function doPost(e) {
  try {
    var body = JSON.parse(e.postData.contents);
    var ss        = SpreadsheetApp.getActiveSpreadsheet();
    var sheetName = body.sheetName || 'Sheet1';
    var mode      = body.mode      || 'append';
    var rows      = body.data      || [];

    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) sheet = ss.insertSheet(sheetName);

    if (mode === 'overwrite') {
      // Hapus semua lalu tulis ulang dari awal
      sheet.clearContents();
      if (rows.length > 0) {
        sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows);
      }
    } else {
      // APPEND — data ditambahkan, header hanya ditulis sekali
      var lastRow = sheet.getLastRow();
      var toWrite = (lastRow > 0) ? rows.slice(1) : rows; // skip header jika sudah ada
      if (toWrite.length > 0) {
        var startRow = (lastRow === 0) ? 1 : lastRow + 1;
        sheet.getRange(startRow, 1, toWrite.length, toWrite[0].length).setValues(toWrite);
      }
      // Auto sort sheet berdasarkan kolom Tanggal (kolom B = index 2)
      if (sheet.getLastRow() > 1) {
        var range = sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn());
        range.sort({column: 2, ascending: true});
      }
    }

    return ContentService
      .createTextOutput(JSON.stringify({status:'ok', mode:mode, rows:rows.length}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error', message:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var action    = (e.parameter && e.parameter.action) ? e.parameter.action : 'ping';
    var sheetName = (e.parameter && e.parameter.sheet)  ? e.parameter.sheet  : 'Sheet1';
    if (action === 'read') {
      var ss    = SpreadsheetApp.getActiveSpreadsheet();
      var sheet = ss.getSheetByName(sheetName);
      if (!sheet) {
        return ContentService
          .createTextOutput(JSON.stringify({status:'error', message:'Sheet "'+sheetName+'" tidak ditemukan. Cek nama sheet di Google Sheet Anda.'}))
          .setMimeType(ContentService.MimeType.JSON);
      }
      var lastRow = sheet.getLastRow();
      var lastCol = sheet.getLastColumn();
      if (lastRow < 1 || lastCol < 1) {
        return ContentService
          .createTextOutput(JSON.stringify({status:'ok', rows:[], sheet:sheetName}))
          .setMimeType(ContentService.MimeType.JSON);
      }
      var rows = sheet.getRange(1, 1, lastRow, lastCol).getValues();
      return ContentService
        .createTextOutput(JSON.stringify({status:'ok', rows:rows, sheet:sheetName, count:rows.length-1}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    return ContentService
      .createTextOutput(JSON.stringify({status:'ready', msg:'HRD Dashboard API aktif — gunakan ?action=read&sheet=NamaSheet untuk baca data'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error', message:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
